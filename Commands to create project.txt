# 1. Create app (no standalone root, SCSS, routing)
ng new advanced-ng20-demo \
  --routing \
  --style=scss \
  --standalone=false \
  --skip-tests

cd advanced-ng20-demo

# Core (interceptors, guards, auth service, etc.)
ng g module core

#Reducers:

ng g class reducers/index

# Shared (pipes, shared components)
ng g module shared

# Dashboard: signals, subjects, new control flow, RxJS demos
ng g module features/dashboard --routing

# Forms: reactive + template-driven
ng g module features/forms --routing

# NgRx demo module
ng g module features/ngrx-demo --routing

# Keep AppComponent as module-based root component (already created)
# Generate one standalone component (ONLY ONE)
ng g component shared/standalone-profile --standalone=true --skip-tests=true --style=scss

#Dashboard components

ng g component features/dashboard/components/signals-demo --module=features/dashboard/dashboard --skip-tests=true --style=scss
ng g component features/dashboard/components/subjects-demo --module=features/dashboard/dashboard --skip-tests=true --style=scss
ng g component features/dashboard/components/rxjs-operators-demo --module=features/dashboard/dashboard --skip-tests=true --style=scss

#Forms components

ng g component features/forms/components/reactive-form-demo --module=features/forms/forms --skip-tests=true --style=scss
ng g component features/forms/components/template-form-demo --module=features/forms/forms --skip-tests=true --style=scss

#NGRX demo components

ng g component features/ngrx-demo/components/ngrx-todos --module=features/ngrx-demo/ngrx-demo --skip-tests=true --style=scss

#Auth visual component (to show MSAL-ish I/O)

ng g component core/components/auth-panel --module=core/core --skip-tests=true --style=scss
